# setting ipmi from the host

service ipmi start
ipmitool lan print

ipmitool lan set 1 ipaddr 1.2.3.4
ipmitool lan set 1 netmask 255.255.255.192
ipmitool lan set 1 defgw ipaddr 1.2.3.1

ipmitool lan print
service ipmi stop
